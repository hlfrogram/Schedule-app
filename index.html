// /api/claude.js // 프론트엔드는 이 엔드포인트만 호출합니다. (예: fetch('/api/claude', {...})) // 실제 Anthropic API 키는 이 서버 코드 안에서만 사용되고, // 브라우저로는 절대 노출되지 않습니다. // // 배포 후 Vercel 프로젝트 설정 > Environment Variables 에서 // ANTHROPIC_API_KEY 값을 반드시 등록해야 합니다. export default async function handler(req, res) { // CORS: 같은 도메인에서만 호출하는 게 기본이지만, // 개발 중 다른 origin에서 테스트할 경우를 대비해 최소한으로 열어둡니다. res.setHeader('Access-Control-Allow-Origin', '*'); res.setHeader('Access-Control-Allow-Methods', 'POST, OPTIONS'); res.setHeader('Access-Control-Allow-Headers', 'Content-Type'); if (req.method === 'OPTIONS') { return res.status(200).end(); } if (req.method !== 'POST') { return res.status(405).json({ error: { message: 'POST만 허용됩니다.' } }); } const apiKey = process.env.ANTHROPIC_API_KEY; if (!apiKey) { return res.status(500).json({ error: { message: '서버에 ANTHROPIC_API_KEY 환경변수가 설정되어 있지 않습니다.' } }); } try { // 프론트에서 보낸 body(model, max_tokens, messages 등)를 그대로 Anthropic에 전달 const { model, max_tokens, messages } = req.body || {}; if (!messages) { return res.status(400).json({ error: { message: 'messages 필드가 필요합니다.' } }); } const upstream = await fetch('https://api.anthropic.com/v1/messages', { method: 'POST', headers: { 'Content-Type': 'application/json', 'x-api-key': apiKey, 'anthropic-version': '2023-06-01' }, body: JSON.stringify({ model: model || 'claude-sonnet-4-6', max_tokens: max_tokens || 1000, messages }) }); const data = await upstream.json(); if (!upstream.ok) { return res.status(upstream.status).json(data); } return res.status(200).json(data); } catch (err) { console.error('claude proxy error:', err); return res.status(500).json({ error: { message: '프록시 서버 오류: ' + err.message } }); } }
